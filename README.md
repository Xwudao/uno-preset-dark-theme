## install

```bash
pnpm i uno-preset-dark-theme -D
```

## config

```ts
export default defineConfig({
  rules: [],
  theme: {
    dark:{
		colors: {
			primary: '#e67e22',
			black: '#17171a',
		},
    },
    light:{
		colors: {
			primary: '#e67e22',
			black: '#17171a',
		},
    }
  },
  presets: [
    presetTheme({
    	element: 'body'
    }),
    // other preset, such as PresetWind
    // but, this preset should be used with other presets
  ],
});

```
